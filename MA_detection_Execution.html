def adjust_gamma(image, gamma=1.0): #To change Contrast and Brightness(if Image is too dark or too Bright/Bleech)

   
   table = np.array([((i / 255.0) ** gamma) * 255
      for i in np.arange(0, 256)]).astype("uint8")

   return cv2.LUT(image, table)
